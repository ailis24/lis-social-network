```javascript
// Загрузка ленты:
const loadFeed = async () => {
  const posts = await db.collection('posts').get();
  
  // Сортировка:
  const sortedPosts = posts.sort((a, b) => {
    // Premium посты выше
    const aScore = a.author.isPremium ? 1000 : 0;
    const bScore = b.author.isPremium ? 1000 : 0;
    
    // Плюс время публикации
    return (bScore + b.createdAt) - (aScore + a.createdAt);
  });
  
  setFeed(sortedPosts);
};
```
